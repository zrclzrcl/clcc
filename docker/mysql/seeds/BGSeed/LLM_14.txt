SET @OLD_SQL_MODE = @@SQL_MODE;
SET SQL_MODE = 'NO_ENGINE_SUBSTITUTION,STRICT_ALL_TABLES';

CREATE TABLE t1 (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    geom GEOMETRY NOT NULL SRID 0,
    dt DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    txt TEXT CHECK (txt REGEXP '[A-Za-z]{3}'),
    num DECIMAL(10,5) UNSIGNED ZEROFILL
);

INSERT INTO t1 (geom, txt, num) VALUES

(ST_GeomFromText('LineString(0 0, 1 1)'), 'XYZ', 500.12345);
